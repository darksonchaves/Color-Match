function getColorName(r, g, b) { const colors = [ { name: "Vermelho", r: 255, g: 0, b: 0 }, { name: "Verde", r: 0, g: 255, b: 0 }, { name: "Azul", r: 0, g: 0, b: 255 }, { name: "Amarelo", r: 255, g: 255, b: 0 }, { name: "Laranja", r: 255, g: 165, b: 0 }, { name: "Roxo", r: 128, g: 0, b: 128 }, { name: "Rosa", r: 255, g: 192, b: 203 }, { name: "Marrom", r: 139, g: 69, b: 19 }, { name: "Branco", r: 255, g: 255, b: 255 }, { name: "Preto", r: 0, g: 0, b: 0 }, { name: "Cinza", r: 128, g: 128, b: 128 } ]; let minDistance = Infinity; let closestColor = "Indefinida"; colors.forEach(color => { const distance = Math.sqrt( (r - color.r) ** 2 + (g - color.g) ** 2 + (b - color.b) ** 2 ); if (distance < minDistance) { minDistance = distance; closestColor = color.name; } }); return closestColor; }
